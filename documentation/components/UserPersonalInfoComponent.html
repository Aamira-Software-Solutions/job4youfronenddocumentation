import { Component, OnInit } from '@angular/core';
import { FormControl, FormGroup,  Validators, FormArray, FormBuilder, Form} from '@angular/forms'
import { DialogService } from 'src/app/services/dialog.service';

import {  UserProfileService } from "src/app/services/user-profile.service"
 
@Component({
  selector: 'app-user-personal-info',
  templateUrl: './user-personal-info.component.html',
  styleUrls: ['./user-personal-info.component.css']
})
export class UserPersonalInfoComponent implements OnInit {

  data: any;
  personalinfoForm: FormGroup;
  username: string;
  relatedList: any = ['personal', 'referee', 'both'];
  submitted = false;

  
  

  constructor(private fb: FormBuilder, private userService: UserProfileService, private confirmDialog: DialogService) {

    this.userService.getPersonalInfo().subscribe(
       (success)=>{
         this.data = success.data
         //console.log(success)
        

         this.setPersonalInfo()
       }, 
       (err)=>{
         console.log(err)
       }
    )

    this.personalinfoForm = this.fb.group({
      name: this.fb.group({
        first: ['', Validators.required],
        last: ['', Validators.required]
      }),
      dob: [''],
      contact:this.fb.group({
        phone:this.fb.array([
          //this.newPhone()
        ]),
        email:this.fb.array([
         //this.newEmail()
        ])
      }),
      password:[''],
      address:this.fb.group({
        street: [''],
        city: [''],
        postal: [''],
        province: [''],
        country: [''],  
      }),
      jobSearching: [''],
      username: ['']
    });
   }

  ngOnInit(): void {
    
  }

  get personalinfoFormControl(){
    return this.personalinfoForm.controls;
  }

  newPhone(): FormGroup{
    return this.fb.group({
      number: ['',Validators.required],
      related: ['',Validators.required]
    })
  }

  get phone(): FormArray{
    return this.personalinfoForm.controls.contact.get("phone") as FormArray;
  }

  newEmail(): FormGroup{
    return this.fb.group({
      address: ['',[Validators.required ,Validators.email]],
      related: ['',Validators.required]
    })
  }

  get email(): FormArray{
    return this.personalinfoForm.controls.contact.get("email") as FormArray;
  }

  addEmail(): void{
    this.email.push(this.newEmail())
  }
  addPhone(): void{
    this.phone.push(this.newPhone())
  }
  removeEmail(i:number){
    this.email.removeAt(i)
  }
  removePhone(i:number){
    this.phone.removeAt(i)
  }

  setPersonalInfo(){
    console.log(this.data)
    if(typeof this.data.name != 'undefined'){
      this.personalinfoForm.controls.name.get("first").setValue(this.data.name.first)
      this.personalinfoForm.controls.name.get("last").setValue(this.data.name.last)
      this.personalinfoForm.controls.dob.setValue(this.data.dob)
      this.personalinfoForm.controls.address.get("street").setValue(this.data.address.street)
      this.personalinfoForm.controls.address.get("city").setValue(this.data.address.city)
      this.personalinfoForm.controls.address.get("postal").setValue(this.data.address.postal)
      this.personalinfoForm.controls.address.get("province").setValue(this.data.address.province)
      this.personalinfoForm.controls.address.get("country").setValue(this.data.address.country)
      this.personalinfoForm.controls.jobSearching.setValue(this.data.jobSearching)
      this.username = this.data.userName
    }
    
    

    let phonesControl = <FormArray>this.personalinfoForm.controls.contact.get("phone");
    if(this.data.contact != null){
      this.data.contact.phone.forEach(x=>{
        phonesControl.push(this.fb.group({
          number: x.number,
          related: x.related
        }))
      })
    }
     
if(this.data.contact != null){
  let emailsControl = <FormArray>this.personalinfoForm.controls.contact.get("email");
  this.data.contact.email.forEach(x=>{
    emailsControl.push(this.fb.group({
      address: x.address,
      related: x.related
    }))
  })
}
    



  }

 
    
  

  savePersonalInfo(){

    this.submitted = true;
    console.log(this.personalinfoForm);
    //console.log(this.email.controls[0].get('address'))

    


        this.userService.sendPersonalInfo(this.personalinfoForm.value).subscribe(
          (success) => {
            console.log(success);
            this.confirmDialog.openConfirmDialog("Personal Information Successfully Updated")
          },
          (err) => {
            console.log(err)
            this.confirmDialog.openConfirmDialog(err.error.message)
          }
        );
   }

}
