import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators, FormArray, FormControl } from '@angular/forms';
import { MatDialogRef } from '@angular/material/dialog';
import { UserDashboardService } from 'src/app/services/user-dashboard.service';

@Component({
  selector: 'app-create-company',
  templateUrl: './create-company.component.html',
  styleUrls: ['./create-company.component.css']
})
export class CreateCompanyComponent implements OnInit {

  companyForm: FormGroup;

  constructor(private fb: FormBuilder, public dialogRef: MatDialogRef<CreateCompanyComponent>, private dashBoard: UserDashboardService) { 
    this.companyForm = this.fb.group({
      name:['',Validators.required],
      businessType:'',
      address:'',
      phones: this.fb.array([
        this.newPhone
      ]),
      emails: this.fb.array([
        this.newEmail
      ]),
      about:'',
      userId:''

    })
  }

  get newPhone(): FormGroup {
    return this.fb.group({
      number: '',
    })
 }

 get newEmail(): FormGroup {
  return this.fb.group({
    address: '',
  })
}

  get phones(): FormArray{
    return this.companyForm.get("phones") as FormArray
  }

  get emails(): FormArray{
    return this.companyForm.get("emails") as FormArray
  }

  addPhone(): void{
    this.phones.push(this.newPhone)
  }

  addEmail(): void{
    this.emails.push(this.newEmail)
  } 

  onSubmit(){
    this.companyForm.get("userId").setValue(this.dashBoard.userId)
    console.log(this.companyForm.value)
    this.dashBoard.addCompany(this.companyForm.value).subscribe(
      success=>{
        console.log(success)
      },
      (err)=>{
        console.log(err)
      }
    )
    
  }

  onClear():void{
    this.companyForm.reset()
  }
  onClose(){
    this.companyForm.reset();
    this.dialogRef.close();
  }

  ngOnInit(): void {
  }

}
