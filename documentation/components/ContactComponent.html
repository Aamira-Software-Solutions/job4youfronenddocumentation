import { Component, OnInit } from '@angular/core';

import { UserDashboardService } from 'src/app/services/user-dashboard.service'
import { DialogService } from 'src/app/services/dialog.service'
import {FormControl, FormArray, FormGroup, FormBuilder, Validators } from '@angular/forms';

import {UserProfileInfo, email, phone} from "src/app/models/user-profile-info"
import { MatTableDataSource } from '@angular/material/table';
import { MatDialog } from '@angular/material/dialog';
import { CreateNewPhoneComponent } from './create-new-phone/create-new-phone.component';
import { CreateNewEmailComponent } from './create-new-email/create-new-email.component'
import { Router } from '@angular/router';

@Component({
  selector: 'app-contact',
  templateUrl: './contact.component.html',
  styleUrls: ['./contact.component.css']
})
export class ContactComponent implements OnInit {

  selected: string;
  phones:any;
  emails:any;
  address:any;
  contactForm: FormGroup;

  contactData: UserProfileInfo;
  emailData: email;
  phoneData: phone;
  emailsData: Array<email>;
  phonesData: Array<phone>

  listDataEmail: MatTableDataSource<any>;
  listDataPhone: MatTableDataSource<any>;


  


  constructor(private dashboardService: UserDashboardService, private fb: FormBuilder, private dialogService: DialogService, public dialog: MatDialog, private router: Router) {
    this.dashboardService.getContacts().subscribe(
      (success)=>{
        console.log(success)
        if(success.data.contact != null){
          this.phones =  success.data.contact.phone;
          this.emails = success.data.contact.email;
          this.listDataEmail = new MatTableDataSource(this.emails)
          this.listDataPhone = new MatTableDataSource(this.phones) 
        }
        
      }
    )

    this.dashboardService.getAddress().subscribe(
      (success)=>{
        this.address = success.data.address
      }
    )    
     
   }

  ngOnInit(): void {
    
    
    
  }
  displayedColumns: string[] = ['address', 'related','actions'];
  displayedColumnsPhone: string[] = ['number', 'related','actions'];

  onPhoneDelete(phone){
    this.dialogService.openConfirmDialog("Sure want to delete?")
    .afterClosed().subscribe(res =>{
      console.log(res)
      if(res){
        this.dashboardService.deletePhone(phone._id).subscribe(
          res =>{
            console.log(res)
            setTimeout(() => {
              this.dashboardService.redirectTo("/userdashboard")
            }, 500);
          }
        )
      }
    });
  }

  onEmailDelete(email){
    this.dialogService.openConfirmDialog("Sure want to delete?")
    .afterClosed().subscribe(res =>{
      console.log(res)
      if(res){
        console.log(email)
        this.dashboardService.deleteEmail(email._id).subscribe(
          res =>{
            console.log(res)
            setTimeout(() => {
              this.dashboardService.redirectTo("/userdashboard")
            }, 500);
          }
        )
      }
    });
  }

  onPhoneEdit(phone){
    this.dialogService.openPhoneEditDialog(phone)
  }

  onEmailEdit(email){
    this.dialogService.openEmailEditDialog(email)
  }

  openPhoneDialog(): void {
    const dialogRef = this.dialog.open(CreateNewPhoneComponent, {
      width: '60%',
    });

    dialogRef.afterClosed().subscribe(result => {
      console.log(result);
    });
  }


  openEmailDialog(): void {
    const dialogRef = this.dialog.open(CreateNewEmailComponent, {
      width: '60%',
    });

    dialogRef.afterClosed().subscribe(result => {
      console.log(result);
    });
  }

  
    

  

  
  
    
   }
