import { Component, OnInit, Inject } from '@angular/core';
import { FormGroup, FormBuilder, Validators } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialogRef } from '@angular/material/dialog';
import { UserDashboardService } from 'src/app/services/user-dashboard.service';

@Component({
  selector: 'app-create-new-email',
  templateUrl: './create-new-email.component.html',
  styleUrls: ['./create-new-email.component.css']
})
export class CreateNewEmailComponent implements OnInit {

  emailForm: FormGroup;
  relatedList: any = ['personal', 'referee', 'both'];

  constructor(@Inject(MAT_DIALOG_DATA) public data, private fb: FormBuilder, public dialogRef: MatDialogRef<CreateNewEmailComponent>, private dashboardService: UserDashboardService) { 
    this.emailForm = this.fb.group({
      contact: this.fb.group({
        email: this.fb.group({
          address:['', [Validators.required, Validators.email]],
          related:'',
          _id:''
        })
      })
    })

    if (this.data){
      this.populateForm(this.data)
    }
  }

  populateForm(contact) {
    this.emailForm.controls.contact.get("email").get("address").setValue(contact.email.address)
    this.emailForm.controls.contact.get("email").get("related").setValue(contact.email.related)
    this.emailForm.controls.contact.get("email").get("_id").setValue(contact.email._id)
  }

  onClear() {
    this.emailForm.reset();
  }

  onSubmit(){
    console.log(this.emailForm.value.contact.email._id)
    console.log(this.emailForm.value)
    //Two function one with id(patch) one without(post)

    //out should use patch
       this.dashboardService.putContact(this.emailForm.value.contact).subscribe(
         (success) => {
           console.log(success);
           this.onClose()
         },
         (err) => {
           console.log(err)
         }
       );

       setTimeout(() => {
        this.dashboardService.redirectTo("/userdashboard")
      }, 500);
    
  }

  ngOnInit(): void {
  }

  onClose() {
    this.emailForm.reset();
    this.dialogRef.close();
  }

}
