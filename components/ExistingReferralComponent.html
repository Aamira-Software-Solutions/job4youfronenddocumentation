import { Component, OnInit } from '@angular/core';
import { UserDashboardService } from 'src/app/services/user-dashboard.service';
import { FormGroup, FormBuilder, FormArray } from '@angular/forms';

@Component({
  selector: 'app-existing-referral',
  templateUrl: './existing-referral.component.html',
  styleUrls: ['./existing-referral.component.css']
})
export class ExistingReferralComponent implements OnInit {


  candidateName:string;
  pendigReferralForm: FormGroup;
  updateDate:string;
  data:any;

  existingData: any;
  existingReferralForm: FormGroup;


  constructor(private dashboardService: UserDashboardService, private fb: FormBuilder) {

    this.pendigReferralForm = this.fb.group({
      requestor:'',
      referee:'',
      skills: this.fb.array([
        
      ])
      
    })

    this.existingReferralForm = this.fb.group({
      requestor:'',
      referee:'',
      skills: this.fb.array([
        
      ])
      
    })

    this.candidateName = this.dashboardService.getCandidate().referredName
     this.dashboardService.getExistingReferral("requestor", "referee").subscribe(
       (success)=>{
         this.existingData = success
         this.populateExistingForm()
       },
       (err)=>{
         console.log(err)
       }
     ) 

    this.dashboardService.getPendingReferral("requestor", "referee").subscribe(
      (success)=>{
        
        // console.log(success)
        this.data = success.data;
        this.updateDate = this.data.updatedAt;
        this.updateDate = this.updateDate.split("T",2)[0]
        this.populateForm()
      },
      (err)=>{
        console.log(err)
      }
    )
   }
   get skills():FormArray{
     return this.pendigReferralForm.get('skills') as FormArray
   }

   get skillsEx():FormArray{
    return this.existingReferralForm.get('skills') as FormArray
  }

   get newSkill():FormGroup{
     return this.fb.group({
       name:'',
       score:'',
       comment:''
     })
   }

   populateForm(){
    this.pendigReferralForm.get('requestor').setValue(this.data.requestor);
    this.pendigReferralForm.get('referee').setValue(this.data.referee);

    let skillsControl = <FormArray>this.pendigReferralForm.get("skills");
     this.data.skills.forEach(x=>{
      skillsControl.push(this.fb.group({
         name: x.name,
         score:'',
         comment:''
       }))
     })
   }

   populateExistingForm(){
    this.existingReferralForm.get('requestor').setValue(this.existingData.requestor);
    this.existingReferralForm.get('referee').setValue(this.existingData.referee);

    let skillsControl = <FormArray>this.existingReferralForm.get("skills");
     this.existingData.skills.forEach(x=>{
      skillsControl.push(this.fb.group({
         name: x.name,
         score:x.score,
         comment:x.comment
       }))
     })
     console.log(skillsControl)
   }

  ngOnInit(): void {

    
    
  }

  onSubmit():void{
    console.log(this.pendigReferralForm.value)
    this.dashboardService.putReferral(this.pendigReferralForm.value).subscribe(
      (success)=>{
        console.log(success)
      },
      (err)=>{
        console.log(err)
      }
    )
  }

  onUpdate():void{
    console.log(this.existingReferralForm.value)
  }



}
