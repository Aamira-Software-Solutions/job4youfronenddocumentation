import { Component, OnInit, Input } from '@angular/core';
import { Router, ActivatedRoute, ParamMap } from '@angular/router';

import { AuthService } from 'src/app/services/auth.service';
import { UserProfileService } from 'src/app/services/user-profile.service';
import { UserDashboardService } from 'src/app/services/user-dashboard.service';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { SignupconfirmService } from 'src/app/services/signupconfirm.service';
import { ErrordialogService } from 'src/app/services/errordialog.service';

@Component({
  selector: 'app-login',
  templateUrl: './login.component.html',
  styleUrls: ['./login.component.css'],
})
export class LoginComponent implements OnInit {
  loggedIn: boolean;
  token: any;
  form: FormGroup;

  constructor(
    private fb: FormBuilder,
    private auth: AuthService,
    private router: Router,
    private userService: UserProfileService,
    private dashboardService: UserDashboardService,
    private signupconfirmService: SignupconfirmService,
    public errorDialogService: ErrordialogService
  ) {
    this.loggedIn = false;

    
  }

  ngOnInit(): void {
    this.form = this.fb.group({
      userName: ['', Validators.required],
      password: ['', Validators.required],
    });
  }

  get loginFormControl(){
    return this.form.controls
  }

  login() {
   
    const val = this.form.value;

    if (!this.form.invalid) {
      this.auth.login(val).subscribe(
        (success) => {
          this.userService.storeUserId(success.data._id);
          this.dashboardService.storeUserId(success.data._id)
          this.dashboardService.setRequestor(this.form.value.userName)
          this.dashboardService.setUserName(this.form.value.userName)
          localStorage.setItem('requestor', this.form.value.userName);
          localStorage.setItem('username', this.form.value.userName);
          this.router.navigate(["../userdashboard"])
          this.auth.setLoginValue();
          this.loggedIn = this.auth.getLoginValue();
        },
        (err) => {
          if (err.error.message === 'NotVerified') {
            //should be checked
            this.signupconfirmService.setResend();
            this.signupconfirmService.setUserName(this.form.value.userName);
            this.router.navigate(['../confirm']);
          }else if(err.status === 404 || err.status === 401){
            var data = {
              "reason": "username or password is incorrect"
            }
            this.errorDialogService.openDialog(data);
          }
        }
      );
    }
  }

  logOut() {
    localStorage.clear();
    const val = this.form.value;


    this.auth.logOut(val).subscribe((success) => {
      console.log(success);
      this.auth.resetLoginValue();
      this.loggedIn = this.auth.getLoginValue();
      setTimeout(() => {
        this.router.navigate(['home']);
      }, 500);
    });
  }

}
