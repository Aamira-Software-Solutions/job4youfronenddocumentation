import { Component, OnInit } from '@angular/core';
import { FormControl, FormGroup,  Validators, FormArray, FormBuilder} from '@angular/forms'

import { UserProfileService } from "src/app/services/user-profile.service"

@Component({
  selector: 'app-user-education',
  templateUrl: './user-education.component.html',
  styleUrls: ['./user-education.component.css']
})
export class UserEducationComponent implements OnInit {

  data: any;
  educationForm: FormGroup;

  constructor(private fb:FormBuilder, private userService: UserProfileService) { 
    
    this.userService.getEducations().subscribe(
      (success) => {
        this.data = success.data.educations
        this.setEducations()
      },
      (err) => {
        console.log(err)
      }
    )

    this.educationForm = this.fb.group({
      educations: this.fb.array([
        //this.newEducation()
      ])
    });

  }

  ngOnInit(): void { 
  }

  get educations() : FormArray{
    return this.educationForm.get('educations') as FormArray
  }

  newEducation(): FormGroup {
    return this.fb.group({
      degree:'',
      institution:'',
      area:'',
      passingYear:''
    })
  }

  addEducations(): void {
    this.educations.push(this.newEducation());
  }

  removeEducation(i:number){
    this.educations.removeAt(i);

  }

  setEducations(){
    let control = <FormArray>this.educationForm.controls.educations;
    this.data.forEach(x=>{
      control.push(this.fb.group({
        degree:x.degree,
      institution:x.institution,
      area:x.area,
      passingYear:x.passingYear
      }))
    })
  }

  saveEducationInfo(){
    this.userService.sendEducation(this.educationForm.value.educations).subscribe(
      (success) => {
        console.log(success);
      },
      (err) => {
        console.log(err.statusText)
      }
    )
  }


}
