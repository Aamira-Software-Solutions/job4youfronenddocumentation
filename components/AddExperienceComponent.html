import { Component, Inject, OnInit } from '@angular/core';
import { FormArray, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { UserProfileService } from 'src/app/services/user-profile.service';

@Component({
  selector: 'app-add-experience',
  templateUrl: './add-experience.component.html',
  styleUrls: ['./add-experience.component.css']
})
export class AddExperienceComponent implements OnInit {

  skillName:any[]=[];
  refereeName:any[]=[];
  skillsForm: FormGroup;
  flagReferee:Boolean= false;

  constructor(@Inject(MAT_DIALOG_DATA) public data,private userService: UserProfileService, private fb:FormBuilder,private _snackBar: MatSnackBar,private dialogRef: MatDialogRef<AddExperienceComponent>) { 
    this.skillsForm = this.fb.group({
      _id: [''],
      name: ['', Validators.required],
      references: this.fb.array([
        //this.newReference()
      ]),
      experience:['',Validators.required]
    })
    
  }

  get references() : FormArray {
    return this.skillsForm.get("references") as FormArray
  }

  newReference(): FormGroup {
    return this.fb.group({
      referee:['', Validators.required],
      relationship:['', Validators.required],
      _id: [''],
      status: "pending"
    })
  }

  ngOnInit(): void {
    console.log(this.data.exp)
    this.data.exp.forEach(element => {
      this.skillName.push(element.name)
    });
  }
  openSnackBar(msg) {
    this._snackBar.open(msg, 'End now', {
      duration: 5500,
    });
  }
  checkSkill(value){
    if(this.skillName.includes(value.toLowerCase())){
      this.openSnackBar("you can not add same skill twice")
      return false
    }else{
      return true
    }
  }
  checkReferee(){
    if(new Set(this.refereeName).size !== this.refereeName.length ){
      this.openSnackBar("you can not add same referee twice")
      return false
    }else{
      return true
    }
  }

  closeDialog(){
    this.dialogRef.close(false);
}

  addReferences(): void {
    this.references.push(this.newReference())
  }
  removeReferences(i:number) {
    this.references.removeAt(i);
  }

  saveSkill(){
    this.refereeName = []
    this.references.value.forEach(element => {
      this.refereeName.push(element.referee)
    });
    if(this.checkSkill(this.skillsForm.value.name) && this.checkReferee()){
      this.dialogRef.close(this.skillsForm.value);
      console.log(this.skillsForm.value)
      this.userService.sendSkills(this.skillsForm.value).subscribe(
               (success) => {
                 console.log(success);
      
               },
               (err) => { 
                 console.log(err)
               }
             )
    }else{
      console.log("error")
    }
  }

}
