import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, FormArray, Validators } from '@angular/forms';
import { UserDashboardService } from 'src/app/services/user-dashboard.service';

@Component({
  selector: 'app-post-job',
  templateUrl: './post-job.component.html',
  styleUrls: ['./post-job.component.css']
})
export class PostJobComponent implements OnInit {

  jobForm: FormGroup;
  companies;

  constructor(private fb:FormBuilder, private dashboardService: UserDashboardService) {
    this.jobForm = this.fb.group({
      title: ['',Validators.required],
      location:this.fb.group({
        city:'',
        province:'',
        post:'',
        country:''
      }),
      salary:this.fb.group({
        min:'',
        max:''
      }),
      company:'',
      jobType:'',
      applicationDeadline:'',
      open:'',
      aboutJob:'',
      requiredSkills:this.fb.array([
        this.newSkill
      ]),
      requiredEducation:this.fb.array([
        this.newEducation
      ]),
      otherNotes:this.fb.array([
        this.newOtherNotes
      ]),
      tests:this.fb.array([
        this.newTest
      ])
    })

    this.dashboardService.getCompany().subscribe(
      (success)=>{
        console.log(success.data.associatedCompany)
        this.companies = success.data.associatedCompany;
      },
      (err)=>{
        console.log(err)
      }
    )
   }

  get requiredSkills() : FormArray {
    return this.jobForm.get("requiredSkills") as FormArray
  }

  get requiredEducation() : FormArray {
    return this.jobForm.get("requiredEducation") as FormArray
  }

  get tests() : FormArray {
    return this.jobForm.get("tests") as FormArray
  }

  get otherNotes(): FormArray{
    return this.jobForm.get("otherNotes") as FormArray
  }
  

  get newSkill():FormGroup{
     return this.fb.group({
       name:'',
       experience:'',
       additionalNote:'',
       mustHave:''
     })
  }

  get newEducation():FormGroup{
     return this.fb.group({
       degree:'',
       area:'',
       additionalNote:''
     })
  }

  get newOtherNotes(): FormGroup{
      return this.fb.group({
        title:'',
        description:''
    })
  }

  get newTest():FormGroup{
     return this.fb.group({
       question:'',
       questionType:''
     })
  }

  addOtherNotes(): void{
    this.otherNotes.push(this.newOtherNotes)
  }

  addSkill(): void{
    this.requiredSkills.push(this.newSkill)
  }

  addEducation(): void{
    this.requiredEducation.push(this.newEducation)
  }

  addQuestion(): void{
    this.tests.push(this.newTest)
  }

   

  ngOnInit(): void {
  }

  onSubmit():void{
    console.log(this.jobForm.value)
    this.dashboardService.addJob(this.jobForm.value).subscribe(
      (success)=>{
        console.log(success)
      },
      (err)=>{
        console.log(err)
      }
    )

  }

  onClear():void{
    this.jobForm.reset()
  }

}
