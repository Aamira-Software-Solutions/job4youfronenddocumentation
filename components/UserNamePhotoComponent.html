import { Component, OnInit, Input } from '@angular/core';
import { Router } from '@angular/router';
import { DialogService } from 'src/app/services/dialog.service';
import { UserProfileService } from 'src/app/services/user-profile.service';

@Component({
  selector: 'app-user-name-photo',
  templateUrl: './user-name-photo.component.html',
  styleUrls: ['./user-name-photo.component.css'],
})
export class UserNamePhotoComponent implements OnInit {
  @Input() caption: string;

  fileName = '';
  filePath: string = 'https://vgraphs.com/images/agents/sage-avatar.jpg'; //
  imageToShow:any;
  public message: string;
  constructor(private router: Router, private userservice: UserProfileService, private confirmDialog: DialogService) {}

  ngOnInit(): void {
    this.getImage();
  }

  getImage(){
    this.userservice.getPhoto().subscribe(
      (success)=>{
        console.log(success)
        this.createImageFromBlob(success)
      },
      (err)=>{
        console.log(err)
      }
    )
  }

  createImageFromBlob(image: Blob) {
    let reader = new FileReader();
    reader.addEventListener("load", () => {
       this.imageToShow = reader.result;
       console.log(this.imageToShow)
    }, false);
 
    if (image) {
       reader.readAsDataURL(image);
    }
 }

  editProfile() {
    this.router.navigate(['../userprofile']);
  }
  onFileSelected(event) {

    const file:File = event.target.files[0];
    var imageType = /image.*/;
    if(file && file.type.match(imageType)){
      this.fileName = file.name;
      this.onUpload(file);
    }
    

    
  }

  onUpload(file) {
    const reader = new FileReader();
    reader.onload = () => {
      this.imageToShow = reader.result as string;
    };
    reader.readAsDataURL(file);


    const fd = new FormData();
    fd.append('Image', file);

    this.userservice.sendPhoto(fd).subscribe(
      res=>{
        this.confirmDialog.openConfirmDialog(res.message)
      },
      err=>{
        console.log(err)
      }
    )
  }

  //backend implementation nedded for help https://www.youtube.com/watch?v=MGAjcZNkl3M
}
