import { Component, OnInit, ViewChild } from '@angular/core';
import { FormControl, FormGroup,  Validators, FormArray, FormBuilder, FormControlName} from '@angular/forms'



import { UserProfileService } from "src/app/services/user-profile.service"
import { Skills } from 'src/app/models/account-credentials'
import { MatSnackBar } from '@angular/material/snack-bar';
import { MatTable, MatTableDataSource } from '@angular/material/table';
import { DialogService } from 'src/app/services/dialog.service';
import { MatDialog, MatDialogRef } from '@angular/material/dialog';
import { EditExperienceComponent } from './edit-experience/edit-experience.component';


@Component({
  selector: 'app-user-skill',
  templateUrl: './user-skill.component.html',
  styleUrls: ['./user-skill.component.css']
})




export class UserSkillComponent implements OnInit {



  data: any;
  skillName:any[]=[];
  refereeName:any[]=[];
  skillsForm: FormGroup;
  skill:Skills[]=[];
  flagSkill:Boolean= false;
  flagReferee:Boolean= false;

  listData: MatTableDataSource<any>;
  displayedColumns: string[] = ['skillName', 'year', 'referee', 'actions'];

  @ViewChild(MatTable) table: MatTable<any>;

  


  constructor(private fb:FormBuilder, private userService: UserProfileService, private _snackBar: MatSnackBar,private dialogService: DialogService,private dialog: MatDialog) {

    this.userService.getSkills().subscribe(
      (success) => {
        this.data = success.data.skills
        //this.setSkills()
        console.log(this.data)
        this.listData = new MatTableDataSource(this.data);
      },
      (err) => { 
        console.log(err)
      }
    )

    this.skillsForm = this.fb.group({
      skills: this.fb.array([])
    })

    this.skill.push(new Skills("java", "test#"))

    console.log(this.skill)

    
    
  }

  ngOnInit(): void {
    //this.table.renderRows();
  }

  editExperience(experience): void {
    let dialogRef = this.dialogService.openEditExperience(experience)
    dialogRef.afterClosed().subscribe(result=>{
      console.log(result)
      if(result != false){
        let index = this.listData.data.indexOf(experience)
        this.listData.data[index] = result
        this.table.renderRows()
      }

    })
  }

  daleteExperience(row){
    console.log(row)
    this.userService.deleteSkill(row._id).subscribe(
      (success)=>{ 
        this.openSnackBar("deleted successfully")
      }
    )
  }
 
 

  

addSkills(): void {
    let dialogRef = this.dialogService.openAddSkill(this.data)
    dialogRef.afterClosed().subscribe(result=>{
      if(result!=false){
        this.listData.data.push(result)
        this.table.renderRows()
      }
      
    })
}


  openSnackBar(msg) {
    this._snackBar.open(msg, 'End now', {
      duration: 5500,
    });
  }

  

}
