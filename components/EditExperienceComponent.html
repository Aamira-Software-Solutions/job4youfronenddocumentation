import { Component, Inject, OnInit } from '@angular/core';
import { FormArray, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { DialogService } from 'src/app/services/dialog.service';
import { UserProfileService } from 'src/app/services/user-profile.service';

@Component({
  selector: 'app-edit-experience',
  templateUrl: './edit-experience.component.html',
  styleUrls: ['./edit-experience.component.css']
})
export class EditExperienceComponent implements OnInit {

  val:any;
  exp:any;
  formGroup:FormGroup;
  expData={
    name: {
      type: String,
      required: true
    },
    experience: {
      type: Number,
    },
    references: [],
    _id:{
      type: String
    }
  }

  constructor(@Inject(MAT_DIALOG_DATA) public data,private fb:FormBuilder,private userService: UserProfileService, private dialogRef: MatDialogRef<EditExperienceComponent>,private confirmDialog: DialogService) { 
  

    this.formGroup = new FormGroup({
      references: this.fb.array([
        //this.newReference()
      ])
    });
 
  }

  ngOnInit(): void {
    console.log(this.data)
    this.val = this.data.experience.experience //exp value 2,3
    this.exp= this.data.experience; //experience object
    this.expData.name = this.data.experience.name
    this.expData._id = this.data.experience._id;
    
  }
  get references() : FormArray {
    return this.formGroup.get("references") as FormArray
  }

  newReference(): FormGroup {
    return this.fb.group({
      referee:['', Validators.required],
      relationship:['', Validators.required],
      _id: [''],
      status: "pending"
    })
  }

  addReferences(): void {
    this.references.push(this.newReference())
  }

  removeReferencesForm(i:number){
    this.references.removeAt(i);
  }
  removeReferences(ref){
    ref.status = "deleted"
  }

  closeDialog(){
    this.dialogRef.close(false)
}


updataeSkill(){
  this.exp.experience = this.val
  this.expData.experience = this.val;
  
  this.expData.references = this.references.value //adding value

  this.data.experience.references.forEach(element => {
    this.expData.references.push(element)
  });

///// this.expData
  console.log(this.expData)
  this.userService.updateSkills(this.expData).subscribe(
    (success) => {
      this.confirmDialog.openConfirmDialog("Experience Successfully Updated")
      console.log(success);
      this.dialogRef.close(this.expData);

    },
    (err) => { 
      console.log(err)
    }
  )
  //console.log(this.expData)
}

}
