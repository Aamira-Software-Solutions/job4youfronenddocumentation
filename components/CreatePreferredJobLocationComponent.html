import { Component, OnInit } from '@angular/core';
import { FormGroup, FormControl, Validators } from "@angular/forms";

@Component({
  selector: 'app-create-preferred-job-location',
  templateUrl: './create-preferred-job-location.component.html',
  styleUrls: ['./create-preferred-job-location.component.css']
})
export class CreatePreferredJobLocationComponent implements OnInit {

  constructor() { }

  form: FormGroup = new FormGroup({
    $key: new FormControl(null),
    country: new FormControl('', Validators.required),
    state: new FormControl('1'),
    city: new FormControl('', [Validators.required, Validators.minLength(4)])
  });

  ngOnInit(): void {
  }

  onClear(): void{
    this.form.reset()
  }

}
