import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { DialogService } from 'src/app/services/dialog.service';
import { ForgetpasswordService } from 'src/app/services/forgetpassword.service';
import { Router, ActivatedRoute, ParamMap } from '@angular/router';

@Component({
  selector: 'app-resetpassword',
  templateUrl: './resetpassword.component.html',
  styleUrls: ['./resetpassword.component.css'],
})
export class ResetpasswordComponent implements OnInit {
  passwordForm: FormGroup;

  submitted = false;
  token: String;
  payLoad;

  constructor(
    private fb: FormBuilder,
    private forgetPasswordService: ForgetpasswordService,
    private confirmDialog: DialogService,
    private router: Router,
    private route: ActivatedRoute
  ) {}

  ngOnInit(): void {
    this.passwordForm = this.fb.group(
      {
        password: ['', Validators.required],
        confirmPassword: ['', Validators.required],
      },
      {
        validator: this.MatchPassword('password', 'confirmPassword'),
      }
    );

    this.route.params.subscribe((params) => {
      this.token = params['token'];
    });
  }
  /**
   *
   * @param password
   * @param confirmPassword
   * @returns null if both are matched or set passwordMismatch = true
   */
  MatchPassword(password: string, confirmPassword: string) {
    return (formGroup: FormGroup) => {
      const passwordControl = formGroup.controls[password];
      const confirmPasswordControl = formGroup.controls[confirmPassword];

      if (!passwordControl || !confirmPasswordControl) {
        return null;
      }

      if (
        confirmPasswordControl.errors &&
        !confirmPasswordControl.errors.passwordMismatch
      ) {
        return null;
      }

      if (passwordControl.value !== confirmPasswordControl.value) {
        confirmPasswordControl.setErrors({ passwordMismatch: true });
      } else {
        confirmPasswordControl.setErrors(null);
      }
    };
  }
  /**
   * for html input control
   */
  get passwordFormControl() {
    return this.passwordForm.controls;
  }

  onSubmit() {
    this.submitted = true;

    if (!this.passwordForm.invalid) {
      //this.passwordForm.removeControl('confirmPassword');

      this.payLoad = {
        password: this.passwordForm.value.password,
        token: this.token,
      };

      console.log(this.payLoad);
      this.forgetPasswordService
        .sendNewPassword(this.payLoad)
        .subscribe((res) => {
          console.log(res);
          const dialogRef = this.confirmDialog.openConfirmDialog(
            "Success!! Don't forget it now. Login again to check"
          );
          dialogRef.afterClosed().subscribe((res) => {
            setTimeout(() => {
              this.router.navigate(['home']);
            }, 2000);
          });
        });
    }
  }
}
