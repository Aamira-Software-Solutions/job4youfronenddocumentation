import { FormGroup, FormBuilder, FormArray, FormControl, Validators } from '@angular/forms';

@Component({
  selector: 'app-user-work-experience',
  templateUrl: './user-work-experience.component.html',
  styleUrls: ['./user-work-experience.component.css']
})
export class UserWorkExperienceComponent implements OnInit {

    ////////////////////////////demo data
    items = [

  constructor(private fb: FormBuilder, private userService: UserProfileService, private confirmDialog: DialogService) { 

    this.userService.getExperiences().subscribe(
      (success) => {
        this.data = success.data.experiences
        
        this.oldArrayLen = this.data.length;
        this.setExperiences()

        //this.newExperience

  disableEnd = true;
  disableEndDate(experience):void{

    
    if(experience.get("current").value){
      experience.get("end").disable();
    }else{
      experience.get("end").enable();
    }
  }
  
  get experienceInfos(): FormArray{
    return this.workExperienceForm.get("experienceInfos") as FormArray
  }
  


// Experience array of sevaral group and one array
  get newExperience(): FormGroup{
    return this.fb.group({
      employerName: '',
      jobTitle: '',
      jobDesc: '',
      skillUsed: this.fb.array([
        this.skillUsed
     ]),
      current: '',
       references: this.fb.array([
          this.references
       ]),
      start: '',
      end: '',
      junk:''
    },{
      validator: this.experienceDateValidate('start','end')
    })
  }

 getEndDate(i){
   if(this.experienceInfos.controls.length != 0){
    return this.experienceInfos.controls[i].get('end');
   }else{
     return null
   }
}
getStartDate(i){
  if(this.experienceInfos.controls.length != 0){
   return this.experienceInfos.controls[i].get('start');
  }else{
    return null
  }
}

get skillUsed(): FormGroup{
  return this.fb.group({
    name:''
  })
}

/**
 * validate end date is greater then start date
 */

 experienceDateValidate(start:string, end:string){
   return (formGroup: FormGroup)=>{
    //  const startControl = formGroup.controls[start];
    //  const endControl = formGroup.controls[end]

    // if(!startControl || !endControl){
    //   return null;
    // }

    // if(startControl.errors && !endControl.errors.notValid){
    //   return null;
    // }

    // const startDate = new Date(start).getTime()
    // const endDate = new Date(end).getTime()

    // if(endDate >= startDate){
    //   endControl.setErrors({notValid:true})
    // } else{
    //   endControl.setErrors(null)
    //}
    let field1Value = formGroup.get(start).value;
    let field2Value = formGroup.get(end).value;

    if (field1Value !== '' && field1Value !== field2Value) {
      return { 'match': `value ${field1Value} is not equal to ${field2Value}` }
    }
    return null;


   }
 }

 endDateError(start, end){
   const startDate = new Date(start.value).getTime()
   const endDate = new Date(end.value).getTime()
    if(endDate <= startDate){
       return true
     } else{
       return false
    }
 }
 

// Reference array with groups
get references(): FormGroup{
  return this.fb.group({
    name: '',
    jobTitle: '',
    email: '',
    relationship:''
  })
}

addExperiences(): void{
  (this.workExperienceForm.get("experienceInfos") as FormArray).push(this.newExperience);
}

addReference(experience): void{
  experience.get('references').push(this.references)
}

addSkill(experience):void{
  experience.get('skillUsed').push(this.skillUsed)
}
removeSkill(experience,j:number):void{
  experience.get('skillUsed').removeAt(j);
}


  removeExperience(i:number){
    (this.workExperienceForm.get("experienceInfos") as FormArray).removeAt(i);
  }

  removeReference(experience,i:number){
    experience.get('references').removeAt(i);
  }

  setExperiences(){
    let control = <FormArray>this.workExperienceForm.controls.experienceInfos;
    this.data.forEach(x=>{
      control.push(this.fb.group({
        employerName: x.employerName,
      jobTitle: x.jobTitle,
      jobDesc: x.jobDesc,
      current: x.current,
      skillUsed: this.setSkillUsed(x),
       references: this.setReferences(x),
      start: x.start,
      end: x.end
      }))
    })
///need modification
    // this.data.skillUsed.forEach(x=>{
    //   this.selected.push({
    //     name: x
    //   })
    // })
  } 
  
  setReferences(x){
    let arr = new FormArray([])
    x.references.forEach(y =>{
      arr.push(this.fb.group({
        name: y.name,
    jobTitle: y.jobTitle,
    email: y.email,
    relationship: y.relationship
      }))
    })
    return arr;
  }

  setSkillUsed(x){
    let arr = new FormArray([])
    x.skillUsed.forEach(y=>{
      arr.push(this.fb.group({
        name: y
      }))
    })

    return arr;
  }

  lockSkill(experience){
    let arr = experience.get("skillUsed")
    this.selected.forEach(y =>{
      arr.push(this.fb.group(
        {name:y.name}
      ))
    })

    this.lock = "Locked";
    experience.get("junk").disable();
    
  }




  saveExperienceInfo(){


    for(var i =0; i< this.workExperienceForm.value.experienceInfos.length; i++){
        var popedData = new Array()
        const len =  this.workExperienceForm.value.experienceInfos[i].skillUsed.length
        for(var j= 0; j< len; j++){
            var data = this.workExperienceForm.value.experienceInfos[i].skillUsed.pop()
            popedData.push(data.name)
            
            if(this.workExperienceForm.value.experienceInfos[i].current === ''){
              this.workExperienceForm.value.experienceInfos[i].current = false
            }
             
        }
        for(var j= 0; j< len; j++){
          var data = popedData.pop()
          this.workExperienceForm.value.experienceInfos[i].skillUsed.push(data)
          }
    }
      console.log("clicked")
    
        if(!this.workExperienceForm.invalid){
          this.userService.sendExperience(this.workExperienceForm.value.experienceInfos).subscribe(
            (success) => {
              console.log(success);
              this.confirmDialog.openConfirmDialog("Experience Successfully Updated")
            },
            (err) => {
              console.log(err)
            }
          )
        }
  }

}
