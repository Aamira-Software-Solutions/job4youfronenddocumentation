import { Component, OnInit } from '@angular/core';
import { FormControl, Validators } from '@angular/forms';
import { DialogService } from 'src/app/services/dialog.service';
import { ForgetpasswordService } from 'src/app/services/forgetpassword.service';

@Component({
  selector: 'app-forgetpassword',
  templateUrl: './forgetpassword.component.html',
  styleUrls: ['./forgetpassword.component.css'],
})
export class ForgetpasswordComponent implements OnInit {
  emailFormcontrol = new FormControl('', [
    Validators.required,
    Validators.email,
  ]);
  submitted = false;

  constructor(
    private forgetPasswordService: ForgetpasswordService,
    private confirmDialog: DialogService
  ) {}

  ngOnInit(): void {}

  /**
   * this function will subcript to a service which send user email to backend
   */
  onSubmit() {
    this.submitted = true;
    if (!this.emailFormcontrol.invalid) {
      this.forgetPasswordService
        .sendEmail(this.emailFormcontrol.value)
        .subscribe(
          (res) => {
            console.log(res);
            this.confirmDialog.openConfirmDialog(
              'Check your mail for reset link'
            );
          },
          (err) => {
            console.log(err);
          }
        );
    }
  }
}
