import { Injectable } from '@angular/core';
import { HttpClient } from "@angular/common/http";
import { Observable } from "rxjs";
import { signupUrl } from "src/app/config/api" ;

@Injectable({
  providedIn: 'root'
})
export class UserProfileService {
  //sharing data to components
  userId: String;


  userPersonalInfoUrl:string ;
  userSkillUrl:string ;
  userExperienceUrl:string ;
  userEducationUrl:string ;

  getPersonalInfoUrl:string;


  constructor(private http:HttpClient) {
   }

  


  

  sendPersonalInfo(personalInfo): Observable<any> {
  
    console.log(personalInfo)
      this.userPersonalInfoUrl = signupUrl+"/"+this.userId;
      return this.http.patch<any>(this.userPersonalInfoUrl, personalInfo);
  }

  getPersonalInfo(): Observable<any>{
    
    return this.http.get<any>(signupUrl+"/"+this.userId);
  }
 
  sendSkills(skill): Observable<any>{
    
    this.userSkillUrl = signupUrl+"/"+this.userId+"/skills";
    return this.http.post<any>(this.userSkillUrl,skill);
  }

  updateSkills(skill): Observable<any>{
    return this.http.put<any>(signupUrl+"/skills",skill);
  }

  deleteSkill(id):Observable<any>{
    return this.http.delete<any>(signupUrl+"/skills/"+id)
  }

  getSkills(): Observable<any>{
    return this.http.get<any>(signupUrl+"/"+this.userId + '?fields=skills');
  }

  sendExperience(experience): Observable<any> {
    
    this.userExperienceUrl = signupUrl+"/"+this.userId+"/experiences";
    console.log(this.userExperienceUrl);
    return this.http.post<any>(this.userExperienceUrl, experience)
  }

  getExperiences(){
    return this.http.get<any>(signupUrl+"/"+this.userId + '?fields=experiences');
  }

  sendEducation(education): Observable<any>{
   
    this.userEducationUrl = signupUrl+"/"+this.userId+"/educations";
    return this.http.post<any>(this.userEducationUrl, education)
  }

  getEducations(){
    return this.http.get<any>(signupUrl+"/"+this.userId + '?fields=educations');
  }

  storeUserId(data) {
    
    this.userId = data;
    return 1;
  }

  getUserId() {
    return this.userId;
  }



  
}
