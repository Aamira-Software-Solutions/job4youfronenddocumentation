import { Injectable } from '@angular/core';
import { HttpClient } from "@angular/common/http";
import { Observable } from "rxjs";
import { signupUrl, proxyUrl } from "src/app/config/api" ;
import { Router } from '@angular/router';

@Injectable({
  providedIn: 'root'
})
export class UserDashboardService {

  userId: String; //Should be change later 5f17aa63b5f36808708a1dab
  candidateData:any; //for referral candidate data
  referee:any;
  requestor:any;
  username:string="";
  

  constructor(private http:HttpClient, private router: Router) { }

  redirectTo(uri:string){  //for refreshing
    this.router.navigateByUrl('/', {skipLocationChange: true}).then(()=>
    this.router.navigate([uri]));
 }

 storeUserId(data) {
    
  this.userId = data;
  return 1;
}

getUserId() {
  return this.userId;
}

 //fo referral data methods
 getCandidate(){
   return this.candidateData;
 } 
 setCandidate(data){
   this.candidateData = data
 }
 setRequestor(requestor){
  this.requestor = requestor;
 }
 setReferee(referee){
  this.referee = referee;
 }
 setUserName(name){
  this.username = name
}
getUserName(){
  return this.username
}
 


  getContacts(): Observable<any>{
    return this.http.get<any>(signupUrl+"?fields=contact");
  }

  getAddress(): Observable<any>{
    return this.http.get<any>(signupUrl+"?fields=address");
  }

  putContact(contact): Observable<any>{ 
    return this.http.put<any>(signupUrl+"/contact",contact);
  }

  deletePhone(id){
    return this.http.delete<any>(signupUrl+"/contact/phone/"+id)
  }

  deleteEmail(id){
    return this.http.delete<any>(signupUrl+"/contact/email/"+id)
  }

  addCompany(company){
    return this.http.post<any>("api/companies", company)
  }

  addJob(job){
    return this.http.post<any>("api/jobs", job)
  }

  getCompany(){
    return this.http.get<any>(signupUrl+"?fields=associatedCompany")
  }

  getPendingReferralUser(){
    return this.http.get<any>(signupUrl+"?fields=referral&status=pending")
  }

  getExistingReferral(requestor, referee){
    return this.http.get<any>(proxyUrl + "referral-doc?requestor="+this.requestor+"&referee="+this.referee)
  }

  getPendingReferral(requestor, referee){
    return this.http.get<any>(proxyUrl + "referral-request?requestor="+this.requestor+"&referee="+this.referee)
  }

  putReferral(data){
    return this.http.put<any>(proxyUrl + "referral-doc", data)
  }

}
