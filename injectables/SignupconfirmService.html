import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { resendVrificationUrl, verificationUrl } from '../config/api';

@Injectable({
  providedIn: 'root'
})
export class SignupconfirmService {

  resend:boolean=false;
  username:string=""


  constructor(private http:HttpClient) { }

  setResend(){
    this.resend = true
  }
  getResend(){
    return this.resend
  }
  setUserName(name){
    this.username = name
  }
  getUserName(){
    return this.username
  }
  signupConfirm(tokenValue){
    return this.http.post(verificationUrl, tokenValue)
  }

  resendConfirmLink(){
    return this.http.get(resendVrificationUrl+this.username)
  }

}
